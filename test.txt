// 辅助函数：解码并编码 Basic Auth
function getAuthHeader(username, password) {
  const credentials = `${username}:${password}`;
  const encoded = btoa(credentials);
  return `Basic ${encoded}`;
}

function decodeBasicAuth(authHeader) {
  if (!authHeader || !authHeader.startsWith('Basic ')) return null;
  try {
    const encoded = authHeader.substring(6);
    const decoded = atob(encoded);
    const [username, password] = decoded.split(':');
    return { username, password };
  } catch (e) {
    return null;
  }
}

function guessContentType(filename) {
  const fileExtension = filename.split('.').pop().toLowerCase();
  const mimeTypes = {
    'sh': 'application/x-sh', 'js': 'application/javascript', 'txt': 'text/plain',
    'html': 'text/html', 'css': 'text/css', 'json': 'application/json',
    'jpg': 'image/jpeg', 'jpeg': 'image/jpeg', 'png': 'image/png', 'gif': 'image/gif',
    'svg': 'image/svg+xml', 'webp': 'image/webp', 'pdf': 'application/pdf',
    'zip': 'application/zip', 'tar': 'application/x-tar', 'gz': 'application/gzip',
    'mp3': 'audio/mpeg', 'mp4': 'video/mp4'
  };
  return mimeTypes[fileExtension] || 'application/octet-stream';
}

function generateToken() {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  let result = '';
  for (let i = 0; i < 24; i++) result += chars.charAt(Math.floor(Math.random() * chars.length));
  return result;
}

async function checkToken(request, env, kv) {
  const url = new URL(request.url);
  const token = url.searchParams.get('token') || request.headers.get('X-Token');
  if (!token) return null;
  if (env.TOKEN) {
    const tokens = env.TOKEN.split(',').map(t => t.trim()).filter(Boolean);
    if (tokens.includes(token)) return { permissions: { download: true, upload: true, delete: true }, fromEnv: true };
  }
  if (kv) {
    try {
      const val = await kv.get('__tokens__/list');
      if (val) {
        const list = JSON.parse(val);
        const found = list.find(t => t.token === token);
        if (found) return found;
      }
    } catch (e) {}
  }
  return null;
}

function isSafePath(filename) {
  if (!filename) return false;
  if (filename.includes('..')) return false;
  if (filename.startsWith('/')) return false;
  return true;
}

function isFileOperation(path, method) {
  const internalAPIs = ['list-keys', 'list-keys-flat', 'list-meta', 'node-stats', 'sync-all'];
  if (method === 'GET' && (path === 'stats' || path === 'list-all')) return true;
  if ((method === 'PUT' && path.startsWith('update/')) ||
      (method === 'POST' && path.startsWith('rename/')) ||
      (method === 'DELETE' && path.startsWith('delete/'))) return true;
  if (method === 'GET' && !internalAPIs.includes(path) && path !== '' && !path.startsWith('proxy/') && !path.startsWith('__info__')) return true;
  if (path.startsWith('proxy/')) {
    const parts = path.split('/');
    const targetPath = parts.slice(2).join('/');
    if ((method === 'PUT' && targetPath.startsWith('update/')) ||
        (method === 'POST' && targetPath.startsWith('rename/')) ||
        (method === 'DELETE' && targetPath.startsWith('delete/'))) return true;
    if (method === 'GET' && !internalAPIs.includes(targetPath) && targetPath !== '') return true;
  }
  return false;
}

function hashString(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash |= 0;
  }
  return Math.abs(hash);
}

async function getAllDomains(env, currentDomain, kv) {
  let all = [currentDomain];
  if (kv) {
    try {
      const val = await kv.get('__settings__/domains');
      if (val) all = [...new Set([currentDomain, ...JSON.parse(val)])];
    } catch (e) {}
  }
  if (all.length === 1 && env.OTHER_DOMAINS) {
    const envDoms = env.OTHER_DOMAINS.split(',').map(d => d.trim()).filter(Boolean);
    if (envDoms.length) {
      all = [...all, ...envDoms];
      try { await kv.put('__settings__/domains', JSON.stringify(all.filter(d => d !== currentDomain))); } catch (e) {}
    }
  }
  return all;
}

async function loadKVCredentials(kv) {
  try {
    const val = await kv.get('__settings__/credentials');
    if (val) return JSON.parse(val);
  } catch (e) {}
  return null;
}

async function checkCredentials(authHeader, env, kv) {
  const creds = decodeBasicAuth(authHeader);
  if (!creds) return false;
  const kvCreds = await loadKVCredentials(kv);
  if (kvCreds) return kvCreds.username === creds.username && kvCreds.password === creds.password;
  if (env.JIANGUO_USERNAME && env.JIANGUO_PASSWORD) {
    const match = env.JIANGUO_USERNAME === creds.username && env.JIANGUO_PASSWORD === creds.password;
    if (match) {
      try { await kv.put('__settings__/credentials', JSON.stringify({ username: creds.username, password: creds.password })); } catch (e) {}
    }
    return match;
  }
  return false;
}

async function syncSettingsToAll(key, data, allDomains, currentDomain, authHeader) {
  const body = JSON.stringify(data);
  await Promise.allSettled(
    allDomains.filter(d => d !== currentDomain).map(d =>
      fetch(`https://${d}/__settings__/${key}`, {
        method: 'PUT', body,
        headers: { 'Authorization': authHeader, 'Content-Type': 'application/json' }
      }).catch(() => {})
    )
  );
}

// 跨域代理请求
function buildProxyUrl(targetDomain, targetPath, url) {
  let search = url.search;
  if (!search.includes('node=')) {
    search = (search ? search + '&' : '?') + 'node=' + encodeURIComponent(targetDomain);
  }
  return `https://${targetDomain}/${targetPath}${search}`;
}

async function proxyGet(targetDomain, targetPath, requestAuthHeader, url) {
  const targetUrl = buildProxyUrl(targetDomain, targetPath, url);
  try {
    const resp = await fetch(targetUrl, { headers: { 'Authorization': requestAuthHeader } });
    return new Response(resp.body, resp);
  } catch (e) {
    return new Response(`代理请求失败: ${e.message}\n`, { status: 500 });
  }
}

async function proxyPut(targetDomain, targetPath, request, requestAuthHeader, url) {
  const targetUrl = buildProxyUrl(targetDomain, targetPath, url);
  try {
    const body = await request.text();
    const resp = await fetch(targetUrl, {
      method: 'PUT', body,
      headers: { 'Authorization': requestAuthHeader, 'Content-Type': request.headers.get('Content-Type') || 'application/octet-stream' }
    });
    return new Response(resp.body, resp);
  } catch (e) {
    return new Response(`代理请求失败: ${e.message}\n`, { status: 500 });
  }
}

async function proxyRequest(targetDomain, targetPath, request, requestAuthHeader, url) {
  if (request.method === 'GET' || request.method === 'HEAD') {
    return proxyGet(targetDomain, targetPath, requestAuthHeader, url);
  }
  const targetUrl = buildProxyUrl(targetDomain, targetPath, url);
  try {
    const body = await request.text();
    const fetchInit = {
      method: request.method,
      headers: { 'Authorization': requestAuthHeader, 'Content-Type': request.headers.get('Content-Type') || 'application/octet-stream' }
    };
    if (body) fetchInit.body = body;
    const resp = await fetch(targetUrl, fetchInit);
    return new Response(resp.body, resp);
  } catch (e) {
    return new Response(`代理请求失败: ${e.message}\n`, { status: 500 });
  }
}

function resolveNode(filename, allDomains, nodeParam) {
  if (nodeParam) {
    if (allDomains.includes(nodeParam)) return nodeParam;
    const idx = parseInt(nodeParam);
    if (!isNaN(idx) && idx >= 0 && idx < allDomains.length) return allDomains[idx];
  }
  return allDomains[hashString(filename) % allDomains.length];
}

// --- 信息文件管理（每个域名一个，记录该域名下的所有文件）---
function getInfoKey(domain) {
  return `__info__/${domain}`;
}

async function loadInfo(MY_SCRIPT_BUCKET, domain) {
  const val = await MY_SCRIPT_BUCKET.get(getInfoKey(domain));
  if (val) {
    try { return JSON.parse(val); } catch (e) {}
  }
  return { updated_at: '1970-01-01T00:00:00Z', files: {} };
}

async function saveInfo(MY_SCRIPT_BUCKET, domain, info) {
  info.updated_at = new Date().toISOString();
  await MY_SCRIPT_BUCKET.put(getInfoKey(domain), JSON.stringify(info));
}

async function addFileToInfo(MY_SCRIPT_BUCKET, domain, filename, size, contentType) {
  const info = await loadInfo(MY_SCRIPT_BUCKET, domain);
  info.files[filename] = { size, content_type: contentType };
  await saveInfo(MY_SCRIPT_BUCKET, domain, info);
  return info;
}

async function removeFileFromInfo(MY_SCRIPT_BUCKET, domain, filename) {
  const info = await loadInfo(MY_SCRIPT_BUCKET, domain);
  delete info.files[filename];
  await saveInfo(MY_SCRIPT_BUCKET, domain, info);
  return info;
}

// 获取所有域名信息文件的合并列表
async function getAllFilesFromInfo(MY_SCRIPT_BUCKET, allDomains) {
  const files = [];
  for (const domain of allDomains) {
    const info = await loadInfo(MY_SCRIPT_BUCKET, domain);
    for (const [filename, f] of Object.entries(info.files)) {
      files.push({ filename, size: f.size, content_type: f.content_type, node: domain });
    }
  }
  return files;
}

// 从信息文件找文件所在节点
async function findFileNodeFromInfo(MY_SCRIPT_BUCKET, filename, allDomains, defaultDomain) {
  for (const domain of allDomains) {
    try {
      const info = await loadInfo(MY_SCRIPT_BUCKET, domain);
      if (info.files && info.files[filename]) return domain;
    } catch (e) { /* 忽略读取失败 */ }
  }
  return defaultDomain;
}

// 从信息文件计算节点大小（负载均衡用）
async function computeNodeSizes(MY_SCRIPT_BUCKET, allDomains) {
  const nodeSizes = {};
  for (const domain of allDomains) {
    nodeSizes[domain] = 0;
  }
  for (const domain of allDomains) {
    const info = await loadInfo(MY_SCRIPT_BUCKET, domain);
    for (const f of Object.values(info.files)) {
      nodeSizes[domain] += f.size || 0;
    }
  }
  return nodeSizes;
}

async function findLightestNode(MY_SCRIPT_BUCKET, allDomains) {
  const nodeSizes = await computeNodeSizes(MY_SCRIPT_BUCKET, allDomains);
  let minNode = allDomains[0];
  let minSize = nodeSizes[minNode] || 0;
  for (const domain of allDomains) {
    const size = nodeSizes[domain] || 0;
    if (size < minSize) {
      minSize = size;
      minNode = domain;
    }
  }
  return minNode;
}

      const creds = sessionStorage.getItem('auth');
      if (creds) {
        opts = opts || {};
        opts.headers = opts.headers || {};

    function setupLogin() {
      document.getElementById('login-btn').onclick = async () => {

    async function checkBootstrap() {
      try {
        const resp = await fetch('/domains');
        const data = await resp.json();
        bootstrapped = data.bootstrapped;
        if (!bootstrapped && data.domains.length <= 1) {
          const domain = prompt('请输入已绑定的域名地址（留空表示本节点为首个节点）：\n例如: backup.example.com');
          if (domain !== null && domain.trim()) {
            const syncResp = await fetch('/sync/pull-from', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({ domain: domain.trim() })
            });
            if (syncResp.ok) {
              showStatus('数据同步成功');
              render();
            } else {
              const err = await syncResp.text();
              showStatus('同步失败: ' + err, true);
            }
          } else {
            // First node, initialize empty domain list
            await fetch('/domains', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({ domain: '' })
            });
            showStatus('已初始化为首个节点');
          }
        }
      } catch (e) { /* 忽略 */ }
    }

    function formatSize(bytes) {
      if (!bytes) return '0B';
      if (bytes > 1048576) return (bytes/1048576).toFixed(1)+'MB';
      if (bytes > 1024) return (bytes/1024).toFixed(1)+'KB';
      return bytes+'B';
    }

    function renderFileList(files) {
      const fileListEl = document.getElementById('file-list');
      fileListEl.innerHTML = '';

      if (!files || files.length === 0) {
        const li = document.createElement('li');
        li.textContent = '暂无文件，点击"上传文件"开始使用';
        li.style.justifyContent = 'center';
        fileListEl.appendChild(li);
        return;
      }

      files.sort((a, b) => a.filename.localeCompare(b.filename));

      files.forEach(item => {
        const li = document.createElement('li');
        li.innerHTML = icons.file;

        const nameSpan = document.createElement('span');
        nameSpan.className = 'item-name';
        nameSpan.textContent = item.filename;
        nameSpan.title = item.filename;
        nameSpan.onclick = (e) => {
          e.stopPropagation();
          if (isTextFile(item.filename)) {
            editFile(item.node, item.filename);
          } else {
            downloadFile(item.node, item.filename);
          }
        };
        li.appendChild(nameSpan);

        const metaSpan = document.createElement('span');
        metaSpan.className = 'item-meta';
        metaSpan.textContent = formatSize(item.size) + ' @ ' + item.node;
        metaSpan.title = '节点: ' + item.node + ' | 大小: ' + formatSize(item.size);
        li.appendChild(metaSpan);

        const actions = document.createElement('div');
        actions.className = 'item-actions';
        
        if (isTextFile(item.filename)) {
          const editBtn = document.createElement('button');
          editBtn.title = '编辑';
          editBtn.innerHTML = icons.edit;
          editBtn.onclick = (e) => { e.stopPropagation(); editFile(item.node, item.filename); };
          actions.appendChild(editBtn);
        } else {
          const blankBtn = document.createElement('button');
          blankBtn.style.visibility = 'hidden';
          blankBtn.innerHTML = icons.edit;
          blankBtn.disabled = true;
          actions.appendChild(blankBtn);
        }

        const renameBtn = document.createElement('button');
        renameBtn.title = '重命名';
        renameBtn.innerHTML = icons.rename;
        renameBtn.onclick = (e) => { e.stopPropagation(); renameFile(item.node, item.filename); };
        actions.appendChild(renameBtn);
        
        const downloadBtn = document.createElement('button');
        downloadBtn.title = '下载';
        downloadBtn.innerHTML = icons.download;
        downloadBtn.onclick = (e) => { e.stopPropagation(); downloadFile(item.node, item.filename); };
        actions.appendChild(downloadBtn);

        const deleteBtn = document.createElement('button');
        deleteBtn.title = '删除';
        deleteBtn.innerHTML = icons.delete;
        deleteBtn.onclick = (e) => { e.stopPropagation(); deleteFile(item.node, item.filename); };
        actions.appendChild(deleteBtn);

        li.appendChild(actions);
        fileListEl.appendChild(li);
      });
    }
    
    function isTextFile(filename) {
      const textExtensions = [
        'txt', 'js', 'json', 'html', 'css', 'md', 'xml', 'yml', 'yaml', 'csv', 'log', 'sh', 'conf',

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);
    const path = url.pathname.substring(1);

    const { MY_SCRIPT_BUCKET } = env;

    const authHeader = request.headers.get('Authorization');
    const basicValid = await checkCredentials(authHeader, env, MY_SCRIPT_BUCKET);
    const tokenInfo = !basicValid ? await checkToken(request, env, MY_SCRIPT_BUCKET) : null;
    const isAdmin = basicValid;

    if (path !== '' && (isFileOperation(path, request.method) ? !isAdmin && !tokenInfo : !isAdmin)) {
      return new Response('未授权访问。\n', { status: 401 });
    }

    const requestAuthHeader = isAdmin ? authHeader : getAuthHeader(
      (await loadKVCredentials(MY_SCRIPT_BUCKET) || { username: env.JIANGUO_USERNAME, password: env.JIANGUO_PASSWORD }).username,
      (await loadKVCredentials(MY_SCRIPT_BUCKET) || { username: env.JIANGUO_USERNAME, password: env.JIANGUO_PASSWORD }).password
    );
    const currentDomain = url.hostname;
    const allDomains = await getAllDomains(env, currentDomain, MY_SCRIPT_BUCKET);

    // --- 代理路由 ---
    if (path.startsWith('proxy/')) {
      const parts = path.split('/');
      const targetDomain = parts[1];
      const targetPath = parts.slice(2).join('/');
      if (!targetDomain || !targetPath) {
        return new Response('代理请求格式错误。\n', { status: 400 });
      }
      return proxyRequest(targetDomain, targetPath, request, requestAuthHeader, url);
    }

    // --- GET /__info__ 返回当前域名自己的信息文件 ---
    if (request.method === 'GET' && path === '__info__') {
      const info = await loadInfo(MY_SCRIPT_BUCKET, currentDomain);
      return new Response(JSON.stringify(info), {
        headers: { 'Content-Type': 'application/json;charset=UTF-8' }
      });
    }

    // --- PUT /__info__/<domain> 接收其他域名同步的信息文件（按时间戳更新）---
    if (request.method === 'PUT' && path.startsWith('__info__/')) {
      const domain = path.substring('__info__/'.length);
      if (!domain) return new Response('错误：未指定域名。\n', { status: 400 });
      const body = await request.text();
      let remoteInfo;
      try { remoteInfo = JSON.parse(body); } catch (e) {
        return new Response('错误：无效的 JSON。\n', { status: 400 });
      }
      const localInfo = await loadInfo(MY_SCRIPT_BUCKET, domain);
      if (remoteInfo.updated_at > localInfo.updated_at) {
        await MY_SCRIPT_BUCKET.put(getInfoKey(domain), body);
      }
      return new Response('ok\n', { status: 200 });
    }

    // --- API 路由 ---
    // 上传/更新
    if (request.method === 'PUT' && path.startsWith('update/')) {
      const filename = decodeURIComponent(path.substring('update/'.length));
      if (!filename) return new Response('错误：未指定文件名。\n', { status: 400 });
      if (!isSafePath(filename)) return new Response('错误：文件名不合法（禁止路径遍历）。\n', { status: 400 });
      if (!isAdmin && tokenInfo && !tokenInfo.permissions.upload) return new Response('无上传权限。\n', { status: 403 });

      const nodeParam = url.searchParams.get('node');
      let targetDomain;
      if (nodeParam) {
        targetDomain = allDomains.includes(nodeParam) ? nodeParam : allDomains[parseInt(nodeParam)];
      } else {
        // 跨节点查找文件位置
        targetDomain = await findFileNodeFromInfo(MY_SCRIPT_BUCKET, filename, allDomains, null);
        if (!targetDomain) {
          targetDomain = await findLightestNode(MY_SCRIPT_BUCKET, allDomains);
        }
      }

      if (targetDomain !== currentDomain) {
        return proxyRequest(targetDomain, path, request, requestAuthHeader, url);
      }

      const bodyText = await request.text();
      const contentType = request.headers.get('Content-Type') || guessContentType(filename);
      const fileSize = bodyText.length;

      await MY_SCRIPT_BUCKET.put(filename, bodyText);

      // 更新本地信息文件
      const info = await addFileToInfo(MY_SCRIPT_BUCKET, currentDomain, filename, fileSize, contentType);

      // 同步信息文件到所有其他节点
      const infoBody = JSON.stringify(info);
      await Promise.allSettled(
        allDomains.filter(d => d !== currentDomain).map(d =>
          fetch(`https://${d}/__info__/${currentDomain}`, {
            method: 'PUT', body: infoBody,
            headers: { 'Authorization': requestAuthHeader, 'Content-Type': 'application/json' }
          }).catch(() => {})
        )
      );

      return new Response(JSON.stringify({ success: true, filename, size: fileSize, node: currentDomain }), {
        headers: { 'Content-Type': 'application/json' }, status: 200
      });
    }

    // 删除
    if (request.method === 'DELETE' && path.startsWith('delete/')) {
      const filename = decodeURIComponent(path.substring('delete/'.length));
      if (!filename) return new Response('错误：未指定文件名。\n', { status: 400 });
      if (!isSafePath(filename)) return new Response('错误：文件名不合法（禁止路径遍历）。\n', { status: 400 });
      if (!isAdmin && tokenInfo && !tokenInfo.permissions.delete) return new Response('无删除权限。\n', { status: 403 });

      const nodeParam = url.searchParams.get('node');
      const targetDomain = resolveNode(filename, allDomains, nodeParam);

      if (targetDomain !== currentDomain) {
        return proxyRequest(targetDomain, path, request, requestAuthHeader, url);
      }

      await MY_SCRIPT_BUCKET.delete(filename);

      // 更新本地信息文件
      const info = await removeFileFromInfo(MY_SCRIPT_BUCKET, currentDomain, filename);

      // 同步信息文件到所有其他节点
      const infoBody = JSON.stringify(info);
      await Promise.allSettled(
        allDomains.filter(d => d !== currentDomain).map(d =>
          fetch(`https://${d}/__info__/${currentDomain}`, {
            method: 'PUT', body: infoBody,
            headers: { 'Authorization': requestAuthHeader, 'Content-Type': 'application/json' }
          }).catch(() => {})
        )
      );

      return new Response(JSON.stringify({ success: true, filename, node: currentDomain }), {
        headers: { 'Content-Type': 'application/json' }, status: 200
      });
    }

    // 重命名
    if (request.method === 'POST' && path.startsWith('rename/')) {
      const oldName = decodeURIComponent(path.substring('rename/'.length));
      if (!oldName) return new Response('错误：未指定文件名。\n', { status: 400 });
      if (!isSafePath(oldName)) return new Response('错误：文件名不合法。\n', { status: 400 });
      if (!isAdmin && tokenInfo && !tokenInfo.permissions.delete) return new Response('无删除权限。\n', { status: 403 });

      const bodyText = await request.text();
      let newName;
      try {
        newName = JSON.parse(bodyText).newname;
      } catch (e) {
        return new Response('错误：请求体需要 JSON {newname: "..."}\n', { status: 400 });
      }
      if (!newName || !isSafePath(newName)) {
        return new Response('错误：新文件名不合法。\n', { status: 400 });
      }

      const nodeParam = url.searchParams.get('node');
      const targetDomain = resolveNode(oldName, allDomains, nodeParam);

      if (targetDomain !== currentDomain) {
        const targetUrl = `https://${targetDomain}/rename/${oldName}${url.search}`;
        try {
          const resp = await fetch(targetUrl, {
            method: 'POST', body: bodyText,
            headers: { 'Authorization': requestAuthHeader, 'Content-Type': 'application/json' }
          });
          return new Response(resp.body, resp);
        } catch (e) {
          return new Response(`代理请求失败: ${e.message}\n`, { status: 500 });
        }
      }
      if (!newName || !isSafePath(newName)) {
        return new Response('错误：新文件名不合法。\n', { status: 400 });
      }

      const content = await MY_SCRIPT_BUCKET.get(oldName, { type: 'text' });
      if (content === null) {
        return new Response(`未找到文件: ${oldName}\n`, { status: 404 });
      }

      await MY_SCRIPT_BUCKET.put(newName, content);
      await MY_SCRIPT_BUCKET.delete(oldName);

      const info = await loadInfo(MY_SCRIPT_BUCKET, currentDomain);
      const oldEntry = info.files[oldName];
      if (oldEntry) {
        delete info.files[oldName];
        info.files[newName] = oldEntry;
      }
      await saveInfo(MY_SCRIPT_BUCKET, currentDomain, info);

      const infoBody = JSON.stringify(info);
      await Promise.allSettled(
        allDomains.filter(d => d !== currentDomain).map(d =>
          fetch(`https://${d}/__info__/${currentDomain}`, {
            method: 'PUT', body: infoBody,
            headers: { 'Authorization': requestAuthHeader, 'Content-Type': 'application/json' }
          }).catch(() => {})
        )
      );

      return new Response(JSON.stringify({ success: true, old_name: oldName, new_name: newName }), {
        headers: { 'Content-Type': 'application/json' }, status: 200
      });
    }

    // 返回本地节点统计（内部端点）
    if (request.method === 'GET' && path === 'node-stats') {
      if (!isAdmin && tokenInfo && !tokenInfo.permissions.download) return new Response('无下载权限。\n', { status: 403 });
      const info = await loadInfo(MY_SCRIPT_BUCKET, currentDomain);
      let totalSize = 0;
      const fileList = [];
      for (const [filename, f] of Object.entries(info.files)) {
        totalSize += f.size || 0;
        fileList.push({ filename, size: f.size, content_type: f.content_type });
      }
      return new Response(JSON.stringify({ domain: currentDomain, fileCount: fileList.length, totalSize, files: fileList }), {
        headers: { 'Content-Type': 'application/json;charset=UTF-8' }
      });
    }

    // 列出所有键（仅文件内容，不含 __info__）
    if (request.method === 'GET' && path === 'list-keys') {
      const localList = await MY_SCRIPT_BUCKET.list();
      const localKeys = localList.keys.filter(k => !k.name.startsWith('__info__/')).map(k => k.name);
      const allFilesByDomain = { [currentDomain]: localKeys };

      const results = await Promise.allSettled(
        allDomains.filter(d => d !== currentDomain).map(domain => 
          fetch(`https://${domain}/list-keys-flat`, { headers: { 'Authorization': requestAuthHeader } })
            .then(res => res.ok ? res.json() : Promise.reject())
            .then(data => ({ domain, keys: data.keys }))
            .catch(() => ({ domain, keys: [] }))
        )
      );
      results.forEach(r => { if (r.status === 'fulfilled') allFilesByDomain[r.value.domain] = r.value.keys; });

      return new Response(JSON.stringify(allFilesByDomain), {
        headers: { 'Content-Type': 'application/json;charset=UTF-8' }
      });
    }
    
    // 只返回本地文件键（不含 __info__）
    if (request.method === 'GET' && path === 'list-keys-flat') {
        const localList = await MY_SCRIPT_BUCKET.list();
        const localKeys = localList.keys.filter(k => !k.name.startsWith('__info__/')).map(k => k.name);
        return new Response(JSON.stringify({ keys: localKeys }), {
            headers: { 'Content-Type': 'application/json;charset=UTF-8' }
        });
    }

    // 统一文件列表（从所有 domain 的信息文件读取）
    if (request.method === 'GET' && path === 'list-all') {
      if (!isAdmin && tokenInfo && !tokenInfo.permissions.download) return new Response('无下载权限。\n', { status: 403 });
      const files = await getAllFilesFromInfo(MY_SCRIPT_BUCKET, allDomains);
      return new Response(JSON.stringify({ files }), {
        headers: { 'Content-Type': 'application/json;charset=UTF-8' }
      });
    }

    // 节点统计（从所有 domain 的信息文件读取）
    if (request.method === 'GET' && path === 'stats') {
      if (!isAdmin && tokenInfo && !tokenInfo.permissions.download) return new Response('无下载权限。\n', { status: 403 });
      let totalSize = 0;
      let totalFiles = 0;
      const nodeMap = {};
      for (const domain of allDomains) {
        const info = await loadInfo(MY_SCRIPT_BUCKET, domain);
        const fileCount = Object.keys(info.files).length;
        if (fileCount > 0) {
          let domainSize = 0;
          for (const f of Object.values(info.files)) domainSize += f.size || 0;
          nodeMap[domain] = { fileCount, totalSize: domainSize };
          totalSize += domainSize;
          totalFiles += fileCount;
        }
      }
      const nodes = Object.entries(nodeMap).map(([domain, data]) => ({ domain, ...data }));
      return new Response(JSON.stringify({
        nodes, total: totalFiles, totalSize,
        bucketCount: nodes.length, totalDomains: allDomains.length
      }), {
        headers: { 'Content-Type': 'application/json;charset=UTF-8' }
      });
    }

    // 全量同步：按时间戳拉取所有节点的信息文件
    if (request.method === 'GET' && path === 'sync-all') {
      let updated = 0, skipped = 0, failed = 0;
      const results = await Promise.allSettled(
        allDomains.filter(d => d !== currentDomain).map(async (domain) => {
          try {
            const resp = await fetch(`https://${domain}/__info__`, {
              headers: { 'Authorization': requestAuthHeader }
            });
            if (!resp.ok) { failed++; return; }
            const remoteInfo = await resp.json();
            const localInfo = await loadInfo(MY_SCRIPT_BUCKET, domain);
            if (remoteInfo.updated_at > localInfo.updated_at) {
              await MY_SCRIPT_BUCKET.put(getInfoKey(domain), JSON.stringify(remoteInfo));
              updated++;
            } else {
              skipped++;
            }
          } catch (e) { failed++; }
        })
      );
      return new Response(JSON.stringify({ updated, skipped, failed, total: allDomains.length }), {
        headers: { 'Content-Type': 'application/json;charset=UTF-8' }
      });
    }

    // 保留向后兼容：list-meta 从信息文件读取
    if (request.method === 'GET' && path === 'list-meta') {
      if (!isAdmin && tokenInfo && !tokenInfo.permissions.download) return new Response('无下载权限。\n', { status: 403 });
      const files = await getAllFilesFromInfo(MY_SCRIPT_BUCKET, allDomains);
      return new Response(JSON.stringify({ files }), {
        headers: { 'Content-Type': 'application/json;charset=UTF-8' }
      });
    }

    // --- 设置管理端点（仅管理员 Basic Auth）---
    if (!isAdmin && (path.startsWith('auth/') || path.startsWith('domains') || path.startsWith('tokens') || path.startsWith('sync/'))) {
      return new Response('未授权访问。\n', { status: 401 });
    }

    // GET /auth/check — 检查认证状态
    if (request.method === 'GET' && path === 'auth/check') {
      return new Response(JSON.stringify({ authenticated: isAdmin || !!tokenInfo, isAdmin }), {
        headers: { 'Content-Type': 'application/json' }
      });
    }

    // PUT /auth/credentials — 修改管理员密码
    if (request.method === 'PUT' && path === 'auth/credentials') {
      const body = JSON.parse(await request.text() || '{}');
      if (!body.username || !body.password) return new Response('需要用户名和密码。\n', { status: 400 });
      const creds = { username: body.username, password: body.password };
      await MY_SCRIPT_BUCKET.put('__settings__/credentials', JSON.stringify(creds));
      await syncSettingsToAll('credentials', creds, allDomains, currentDomain, requestAuthHeader);
      return new Response(JSON.stringify({ success: true }), { headers: { 'Content-Type': 'application/json' } });
    }

    // GET /domains — 域名列表
    if (request.method === 'GET' && path === 'domains') {
      const kvVal = await MY_SCRIPT_BUCKET.get('__settings__/domains');
      const kvDoms = kvVal ? JSON.parse(kvVal) : [];
      return new Response(JSON.stringify({ domains: allDomains, kvDomains: kvDoms, bootstrapped: !!kvVal }), {
        headers: { 'Content-Type': 'application/json' }
      });
    }

    // POST /domains — 添加域名（仅首次）
    if (request.method === 'POST' && path === 'domains') {
      const existing = await MY_SCRIPT_BUCKET.get('__settings__/domains');
      if (existing) return new Response('域名列表已锁定，不可手动添加。\n', { status: 400 });
      const body = JSON.parse(await request.text() || '{}');
      const newDomain = body.domain || currentDomain;
      const newDoms = [newDomain];
      await MY_SCRIPT_BUCKET.put('__settings__/domains', JSON.stringify(newDoms));
      const updatedDoms = await getAllDomains(env, currentDomain, MY_SCRIPT_BUCKET);
      await syncSettingsToAll('domains', newDoms, updatedDoms, currentDomain, requestAuthHeader);
      return new Response(JSON.stringify({ success: true, domains: updatedDoms }), {
        headers: { 'Content-Type': 'application/json' }
      });
    }

    // DELETE /domains/<domain> — 删除域名（仅下次自动清理，不可手动）
    if (request.method === 'DELETE' && path.startsWith('domains/')) {
      return new Response('域名不可手动删除。\n', { status: 400 });
    }

    // GET /tokens — 令牌列表
    if (request.method === 'GET' && path === 'tokens') {
      const val = await MY_SCRIPT_BUCKET.get('__tokens__/list');
      const list = val ? JSON.parse(val) : [];
      return new Response(JSON.stringify(list.map((t, i) => ({ ...t, index: i, token: t.token.slice(0, 4) + '***' + t.token.slice(-4) }))), {
        headers: { 'Content-Type': 'application/json' }
      });
    }

    // POST /tokens — 创建令牌
    if (request.method === 'POST' && path === 'tokens') {
      const body = JSON.parse(await request.text() || '{}');
      const permissions = body.permissions || { download: false, upload: false, delete: false };
      const newToken = {
        token: generateToken(),
        name: body.name || '',
        permissions,
        created_at: new Date().toISOString()
      };
      const val = await MY_SCRIPT_BUCKET.get('__tokens__/list');
      const list = val ? JSON.parse(val) : [];
      list.push(newToken);
      await MY_SCRIPT_BUCKET.put('__tokens__/list', JSON.stringify(list));
      await syncSettingsToAll('tokens', list, allDomains, currentDomain, requestAuthHeader);
      return new Response(JSON.stringify({ success: true, token: newToken }), {
        headers: { 'Content-Type': 'application/json' }
      });
    }

    // PUT /tokens/<index> — 修改令牌
    if (request.method === 'PUT' && path.startsWith('tokens/')) {
      const idx = parseInt(path.substring('tokens/'.length));
      const body = JSON.parse(await request.text() || '{}');
      const val = await MY_SCRIPT_BUCKET.get('__tokens__/list');
      const list = val ? JSON.parse(val) : [];
      if (idx < 0 || idx >= list.length) return new Response('令牌不存在。\n', { status: 404 });
      if (body.name !== undefined) list[idx].name = body.name;
      if (body.permissions) list[idx].permissions = body.permissions;
      await MY_SCRIPT_BUCKET.put('__tokens__/list', JSON.stringify(list));
      await syncSettingsToAll('tokens', list, allDomains, currentDomain, requestAuthHeader);
      return new Response(JSON.stringify({ success: true }), { headers: { 'Content-Type': 'application/json' } });
    }

    // DELETE /tokens/<index> — 删除令牌
    if (request.method === 'DELETE' && path.startsWith('tokens/')) {
      const idx = parseInt(path.substring('tokens/'.length));
      const val = await MY_SCRIPT_BUCKET.get('__tokens__/list');
      const list = val ? JSON.parse(val) : [];
      if (idx < 0 || idx >= list.length) return new Response('令牌不存在。\n', { status: 404 });
      list.splice(idx, 1);
      await MY_SCRIPT_BUCKET.put('__tokens__/list', JSON.stringify(list));
      await syncSettingsToAll('tokens', list, allDomains, currentDomain, requestAuthHeader);
      return new Response(JSON.stringify({ success: true }), { headers: { 'Content-Type': 'application/json' } });
    }

    // POST /sync/pull-from — 从已有域名拉取全量数据
    if (request.method === 'POST' && path === 'sync/pull-from') {
      const body = JSON.parse(await request.text() || '{}');
      if (!body.domain) return new Response('需要域名。\n', { status: 400 });
      try {
        const resp = await fetch(`https://${body.domain}/__settings__/all`, {
          headers: { 'Authorization': requestAuthHeader }
        });
        if (!resp.ok) return new Response('同步失败：目标节点拒绝访问。\n', { status: 502 });
        const data = await resp.json();
        if (data.credentials) await MY_SCRIPT_BUCKET.put('__settings__/credentials', JSON.stringify(data.credentials));
        if (data.tokens) await MY_SCRIPT_BUCKET.put('__tokens__/list', JSON.stringify(data.tokens));
        if (data.domains) {
          const existing = await MY_SCRIPT_BUCKET.get('__settings__/domains');
          const merged = [...new Set([...(existing ? JSON.parse(existing) : []), ...data.domains])];
          await MY_SCRIPT_BUCKET.put('__settings__/domains', JSON.stringify(merged));
        }
        if (data.info) {
          for (const [key, val] of Object.entries(data.info)) {
            await MY_SCRIPT_BUCKET.put(key, val);
          }
        }
        const newDomains = await getAllDomains(env, currentDomain, MY_SCRIPT_BUCKET);
        await Promise.allSettled(
          newDomains.filter(d => d !== currentDomain).map(d =>
            fetch(`https://${d}/__settings__/domains`, {
              method: 'PUT',
              body: JSON.stringify([currentDomain]),
              headers: { 'Authorization': requestAuthHeader, 'Content-Type': 'application/json' }
            }).catch(() => {})
          )
        );
        return new Response(JSON.stringify({ success: true, domains: newDomains }), {
          headers: { 'Content-Type': 'application/json' }
        });
      } catch (e) {
        return new Response(`同步失败: ${e.message}\n`, { status: 502 });
      }
    }

    // GET /__settings__/all — 导出全量设置（用于同步）
    if (request.method === 'GET' && path === '__settings__/all') {
      const infoKeys = await MY_SCRIPT_BUCKET.list({ prefix: '__info__/' });
      const info = {};
      for (const key of infoKeys.keys) {
        const val = await MY_SCRIPT_BUCKET.get(key.name);
        if (val) info[key.name] = val;
      }
      const credentials = await loadKVCredentials(MY_SCRIPT_BUCKET);
      const tokensVal = await MY_SCRIPT_BUCKET.get('__tokens__/list');
      const domainsVal = await MY_SCRIPT_BUCKET.get('__settings__/domains');
      return new Response(JSON.stringify({
        credentials,
        tokens: tokensVal ? JSON.parse(tokensVal) : [],
        domains: domainsVal ? JSON.parse(domainsVal) : [],
        info
      }), { headers: { 'Content-Type': 'application/json' } });
    }

    // PUT /__settings__/<type> — 接收同步（内部使用）
    if (request.method === 'PUT' && path.startsWith('__settings__/')) {
      const key = path.substring('__settings__/'.length);
      const body = await request.text();
      if (['credentials', 'tokens', 'domains'].includes(key)) {
        if (key === 'domains') {
          const existing = await MY_SCRIPT_BUCKET.get('__settings__/domains');
          const newDoms = JSON.parse(body);
          const merged = [...new Set([...(existing ? JSON.parse(existing) : []), ...newDoms])];
          await MY_SCRIPT_BUCKET.put('__settings__/domains', JSON.stringify(merged));
        } else if (key === 'tokens') {
          const existing = await MY_SCRIPT_BUCKET.get('__tokens__/list');
          const remote = JSON.parse(body);
          const local = existing ? JSON.parse(existing) : [];
          for (const rt of remote) {
            const found = local.findIndex(l => l.token === rt.token);
            if (found >= 0) local[found] = rt;
            else local.push(rt);
          }
          await MY_SCRIPT_BUCKET.put('__tokens__/list', JSON.stringify(local));
        } else {
          await MY_SCRIPT_BUCKET.put(`__settings__/${key}`, body);
        }
      }
      return new Response('ok\n', { status: 200 });
    }

    // --- UI 页面 ---
    if (path === '') {
      return new Response(fileManagerHtml, {
        headers: { 'Content-Type': 'text/html;charset=UTF-8' }
      });
    }

    // --- 文件下载（从信息文件找所在节点）---
    const nodeParam = url.searchParams.get('node');
    const decodedPath = decodeURIComponent(path);
    if (!isAdmin && tokenInfo && !tokenInfo.permissions.download) {
      // Allow access to settings page HTML and auth check without download permission
      if (decodedPath !== 'auth/check') return new Response('无下载权限。\n', { status: 403 });
    }
    if (nodeParam) {
      const targetDomain = allDomains.includes(nodeParam) ? nodeParam : null;
      if (targetDomain && targetDomain !== currentDomain) {
        return proxyGet(targetDomain, decodedPath, requestAuthHeader, url);
      }
    } else {
      const fileNode = await findFileNodeFromInfo(MY_SCRIPT_BUCKET, decodedPath, allDomains, currentDomain);
      if (fileNode !== currentDomain) {
        return proxyGet(fileNode, decodedPath, requestAuthHeader, url);
      }
    }

    const file = await MY_SCRIPT_BUCKET.get(decodedPath, { type: 'stream' });
    if (file === null) {
      return new Response(`未找到文件: ${decodedPath}\n`, { status: 404 });
    }
    return new Response(file, {
      headers: {
        'Content-Type': guessContentType(decodedPath),
        'Content-Disposition': `attachment; filename*=UTF-8''${encodeURIComponent(decodedPath.split('/').pop())}`
      }
    });
  },

  // 定时任务：同步各节点信息文件
  async scheduled(event, env, ctx) {
    const { MY_SCRIPT_BUCKET } = env;
    if (!MY_SCRIPT_BUCKET) return;
    const kvCreds = await loadKVCredentials(MY_SCRIPT_BUCKET);
    const username = (kvCreds && kvCreds.username) || env.JIANGUO_USERNAME;
    const password = (kvCreds && kvCreds.password) || env.JIANGUO_PASSWORD;
    if (!username || !password) return;
    const requestAuthHeader = getAuthHeader(username, password);
    const allDomains = await getAllDomains(env, 'localhost', MY_SCRIPT_BUCKET);
    
    await Promise.allSettled(
      allDomains.filter(d => d !== 'localhost').map(async (domain) => {
        try {
          const resp = await fetch(`https://${domain}/__info__`, {
            headers: { 'Authorization': requestAuthHeader }
          });
          if (!resp.ok) return;
          const remoteInfo = await resp.json();
          const localInfo = await loadInfo(MY_SCRIPT_BUCKET, domain);
          if (remoteInfo.updated_at > localInfo.updated_at) {
            await MY_SCRIPT_BUCKET.put(getInfoKey(domain), JSON.stringify(remoteInfo));
          }
        } catch (e) {}
      })
    );
  }
};
